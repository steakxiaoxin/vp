    # 一、在文本框光标处插入文本

    利用HTMLInputElement的selectionStart, selectionEnd,和 selectionDirection 属性

    selectionStart: 被选中的第一个字符的位置。

    selectionEnd: 被选中的最后一个字符的 下一个 位置。

    # HTML code

    <textarea id="textarea" rows="5" cols="50"></textarea>
<button id="button">插入一个随机值</button>
<script>
  const textarea = document.getElementById("textarea");
  const button = document.getElementById("button");

  button.addEventListener(
    "click",
    () => {
      const posS = textarea.selectionStart;
      const posE = textarea.selectionEnd;
      // 光标处插入 4 位数随机值
      const insertValue = `{${Math.random()
        .toString()
        .slice(2, 6)}}`;
      textarea.value =
        textarea.value.substring(0, posS) +
        insertValue +
        textarea.value.substring(posE, textarea.value.length);

      // 定位新的光标位置
      textarea.selectionStart = textarea.selectionEnd =
        posS + insertValue.length;
      textarea.focus();
    },
    false
  );
</script>

    # vue 中实现

    <template>
  <div class="el-textarea">
    <textarea v-model="content" id="textarea" type="textarea"></textarea>
    <button @click="insertAtCursor('<张三>')" type="button">插入会员名</button>
  </div>
</template>
<script>
export default {
  data() {
    return {
      content: ""
    };
  },
  methods: {
    async insertAtCursor(myValue) {
      const myField = document.querySelector("#textarea");
      if (myField.selectionStart || myField.selectionStart === 0) {
        var startPos = myField.selectionStart;
        var endPos = myField.selectionEnd;
        this.content =
          myField.value.substring(0, startPos) +
          myValue +
          myField.value.substring(endPos, myField.value.length);
        await this.$nextTick(); // 这句是重点, 圈起来
        myField.focus();
        myField.setSelectionRange(
          endPos + myValue.length,
          endPos + myValue.length
        );
      } else {
        this.content += myValue;
      }
    }
  }
};
</script>

    # 二、嗖的hover一下

    <!DOCTYPE html>
<html lang="en">
  <head>
    <style>
      .hover-underline-animation {
        display: inline-block;
        position: relative;
        color: #0087ca;
      }
      .hover-underline-animation::after {
        content: "";
        position: absolute;
        width: 100%;
        transform: scaleX(0);
        height: 2px;
        bottom: 0;
        left: 0;
        background-color: #0087ca;
        transform-origin: bottom right;
        transition: transform 0.25s ease-out;
      }
      .hover-underline-animation:hover::after {
        transform: scaleX(1);
        transform-origin: bottom left;
      }
    </style>
  </head>
  <body>
    <p class="hover-underline-animation">Hover this text to see the effect!</p>
  </body>
</html>
